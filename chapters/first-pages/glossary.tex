\newglossaryentry{dc}{
    name={Data Center},
    description={
        O local onde estão concentrados os recursos de processamento e armazenamento de dados de uma organização.
    }
}

\newglossaryentry{cluster}{
    name=cluster,
    description={
        Um \textit{cluster} é um conjunto de computadores interligados que trabalham em conjunto para resolver problemas complexos.
    }
}

\newglossaryentry{flavour}{
    name=flavour,
    description={
        \textit{Flavour} é o termo utilizado para referir a configuração específica de uma máquina virtual (memória, CPU, disco, etc).
    }
}
