\addcontentsline{toc}{chapter}{Glossário}
\newglossaryentry{dc}{
    name=Data Center,
    description={
        \textit{dc} é uma abreviatura de \textit{Data Center}, que se refere a um local onde estão
        concentrados os recursos de processamento e armazenamento de dados de uma organização.
    }
}

\newglossaryentry{cluster}{
    name=cluster,
    description={
        Um \textit{cluster} é um conjunto de computadores interligados que trabalham em conjunto
        para resolver problemas complexos.
    }
}
