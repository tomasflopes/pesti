\newglossaryentry{dc}{
    name={data center},
    description={
        O local onde estão concentrados os recursos de processamento e armazenamento de dados de uma organização.
    }
}

\newglossaryentry{cluster}{
    name=cluster,
    description={
        Um conjunto de computadores interligados que trabalham em conjunto para resolver problemas complexos.
    }
}

\newglossaryentry{flavour}{
    name=flavour,
    description={
        A configuração específica de uma máquina virtual (memória, CPU, disco, etc).
    }
}

\newglossaryentry{odd}{
    name=odd,
    description={
        No contexto de apostas desportivas, o valor que é pago ao apostador caso a sua aposta seja vencedora.
    }
}
